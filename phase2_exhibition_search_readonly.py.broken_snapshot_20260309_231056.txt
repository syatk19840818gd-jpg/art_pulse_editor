def _path_mtime_ns(path: Path) -> int:
    try:
        return path.stat().st_mtime_ns
    except OSError:
        return -1


def _normalize_key(text: str) -> str:
    return re.sub(r"\s+", " ", unicodedata.normalize("NFKC", text or "").strip()).casefold()


def _tokenize_semantic(text: str) -> List[str]:
    value = _normalize_key(text)
    if not value:
        return []
    tokens = re.findall(r"[a-z0-9]{2,}|[\u3040-\u30ff\u4e00-\u9fff]{2,}", value)
    return [t for t in tokens if t]


def _truncate_text(text: str, limit: int) -> str:
    value = (text or "").strip()
    if len(value) <= limit:
        return value
    head = value[:limit]
    cut = max(head.rfind("\u3002"), head.rfind("\u3001"), head.rfind(" "), head.rfind("\n"))
    if cut >= int(limit * 0.7):
        return head[: cut + 1].rstrip()
    if limit <= 1:
        return head[:limit]
    return head[: limit - 1].rstrip() + "\u2026"

def _clean_artist_name(value: str) -> str:
    text = re.sub(r"\s+", " ", (value or "").strip())
    text = re.sub(r"(?i)^\s*participating artists?\s*:\s*", "", text).strip()
    text = re.sub(r"(?i)^\s*artists?\s*:\s*", "", text).strip()
    text = re.sub(r"\s+\d+$", "", text).strip()
    return text


def _guess_kana(name_en: str) -> str:
    words = re.findall(r"[A-Za-z]+", name_en or "")
    if not words:
        return ""
    converted: List[str] = []
    for word in words:
        kana = _WORD_KANA_MAP.get(word.upper())
        if not kana:
            return ""
        converted.append(kana)
    return "郢晢ｽｻ".join(converted)


def _google_image_search_url(name_en: str) -> str:
    query = f"{(name_en or '').strip()} art"
    return f"https://www.google.com/search?tbm=isch&q={quote_plus(query)}"


def _build_artist_link_items(raw_artists: str) -> List[dict]:
    cleaned = _clean_artist_name(raw_artists or "")
    parts = [p.strip() for p in re.split(r"[,;\n/|]+", cleaned) if p and p.strip()]
    seen = set()
    rows: List[dict] = []
    for part in parts:
        name_en = _clean_artist_name(part)
        if not name_en:
            continue
        norm = _normalize_key(name_en)
        if norm in seen:
            continue
        seen.add(norm)
        kana = _guess_kana(name_en) or name_en
        rows.append(
            {
                "name_en": name_en,
                "name_kana": kana,
                "google_image_url": _google_image_search_url(name_en),
            }
        )
    return rows


def _build_artist_source_index() -> Tuple[Dict[str, Dict[str, List[str]]], List[str]]:
    warnings: List[str] = []
    index: Dict[str, Dict[str, List[str]]] = {"frieze_london": {}, "liste": {}}
    for fair_slug, path in ARTISTS_TEXT_PATHS.items():
        rows, row_warnings = safe_load_jsonl(path)
        warnings.extend(row_warnings)
        fair_map = index.setdefault(fair_slug, {})
        for row in rows:
            source_url = str(row.get("source_url") or "").strip()
            if not source_url:
                continue
            artist_name = derive_artist_name(source_url, str(row.get("artist_name_en") or ""))
            key = _normalize_key(_clean_artist_name(artist_name))
            if not key:
                continue
            fair_map.setdefault(key, [])
            if source_url not in fair_map[key]:
                fair_map[key].append(source_url)
    return index, warnings


def _build_image_meta_index() -> Tuple[Dict[str, List[dict]], List[str]]:
    warnings: List[str] = []
    image_by_source: Dict[str, List[dict]] = {}
    for fair_slug, path in EXHIBITIONS_IMAGE_META_PATHS.items():
        rows, row_warnings = safe_load_jsonl(path)
        warnings.extend(row_warnings)
        fair_label = FAIR_SLUG_TO_LABEL.get(fair_slug, fair_slug)
        for row in rows:
            source_url = normalize_url(str(row.get("source_url") or ""))
            if not source_url:
                continue
            image_by_source.setdefault(source_url, []).append(
                {
                    "kind": "exhibition",
                    "fair": fair_label,
                    "gallery": str(row.get("gallery_name_en") or ""),
                    "source_url": str(row.get("source_url") or ""),
                    "local_path": str(row.get("local_path") or ""),
                    "image_url": str(row.get("image_url") or ""),
                    "alt": "exhibition image",
                }
            )
    return image_by_source, warnings


@dataclass
class ExhibitionSearchData:
    records: List[dict]
    warnings: List[str]
    total_rows: int
    fair_rows: Dict[str, int]


@lru_cache(maxsize=4)
def _load_exhibition_records_cached(
    frieze_text_mtime_ns: int,
    liste_text_mtime_ns: int,
    frieze_image_mtime_ns: int,
    liste_image_mtime_ns: int,
    frieze_artist_text_mtime_ns: int,
    liste_artist_text_mtime_ns: int,
) -> ExhibitionSearchData:
    del (
        frieze_text_mtime_ns,
        liste_text_mtime_ns,
        frieze_image_mtime_ns,
        liste_image_mtime_ns,
        frieze_artist_text_mtime_ns,
        liste_artist_text_mtime_ns,
    )

    warnings: List[str] = []
    records: List[dict] = []
    fair_rows: Dict[str, int] = {"frieze_london": 0, "liste": 0}

    image_by_source, image_warnings = _build_image_meta_index()
    warnings.extend(image_warnings)
    artist_source_index, artist_source_warnings = _build_artist_source_index()
    warnings.extend(artist_source_warnings)

    for fair_slug, text_path in EXHIBITIONS_TEXT_PATHS.items():
        text_rows, text_warnings = safe_load_jsonl(text_path)
        warnings.extend(text_warnings)
        fair_rows[fair_slug] = len(text_rows)
        fair_label = FAIR_SLUG_TO_LABEL.get(fair_slug, fair_slug)
        fair_artist_index = artist_source_index.get(fair_slug, {})

        for idx, row in enumerate(text_rows, start=1):
            source_url = str(row.get("source_url") or "").strip()
            norm_source_url = normalize_url(source_url)
            artist_names = str(row.get("participating_artists") or "")
            artist_link_items = _build_artist_link_items(artist_names)
            text_value = str(
                row.get("text")
                or row.get("description")
                or row.get("body")
                or row.get("content")
                or ""
            )

            artist_source_urls: List[str] = []
            for item in artist_link_items:
                key = _normalize_key(_clean_artist_name(str(item.get("name_en") or "")))
                artist_source_urls.extend(fair_artist_index.get(key, []))
            artist_source_urls = list(dict.fromkeys([u for u in artist_source_urls if u]))

            all_images = list(image_by_source.get(norm_source_url, []) or [])
            records.append(
                {
                    "id": f"{fair_slug}:{idx}",
                    "fair_slug": fair_slug,
                    "fair_label": fair_label,
                    "gallery_name": str(row.get("gallery_name_en") or ""),
                    "exhibition_title": derive_exhibition_title(row),
                    "year": row.get("target_year"),
                    "artist_names": artist_names,
                    "artist_link_items": artist_link_items,
                    "source_url": source_url,
                    "text": text_value,
                    "summary_ja": str(row.get("summary_ja") or ""),
                    "image_count_hint": len(all_images),
                    "reference_images": all_images,
                    "artist_source_urls": artist_source_urls,
                }
            )

    return ExhibitionSearchData(
        records=records,
        warnings=sorted(set(warnings)),
        total_rows=len(records),
        fair_rows=fair_rows,
    )


def load_exhibition_records_readonly() -> ExhibitionSearchData:
    return _load_exhibition_records_cached(
        _path_mtime_ns(EXHIBITIONS_TEXT_PATHS["frieze_london"]),
        _path_mtime_ns(EXHIBITIONS_TEXT_PATHS["liste"]),
        _path_mtime_ns(EXHIBITIONS_IMAGE_META_PATHS["frieze_london"]),
        _path_mtime_ns(EXHIBITIONS_IMAGE_META_PATHS["liste"]),
        _path_mtime_ns(ARTISTS_TEXT_PATHS["frieze_london"]),
        _path_mtime_ns(ARTISTS_TEXT_PATHS["liste"]),
    )


def _semantic_score(row: dict, query_tokens: List[str], raw_query: str) -> float:
    title = str(row.get("exhibition_title") or "")
    gallery = str(row.get("gallery_name") or "")
    artists = str(row.get("artist_names") or "")
    text = str(row.get("text") or "")[:2500]
    source_url = str(row.get("source_url") or "")
    haystack = " ".join([title, gallery, artists, text, source_url])
    hay_norm = _normalize_key(haystack)
    score = 0.0
    if raw_query and raw_query in hay_norm:
        score += 8.0
    for token in query_tokens:
        if token in _normalize_key(title):
            score += 3.2
        if token in _normalize_key(gallery):
            score += 2.8
        if token in _normalize_key(artists):
            score += 2.4
        if token in hay_norm:
            score += 1.2
    score += min(1.5, len(query_tokens) * 0.08)
    return score


def search_exhibitions(
    records: List[dict],
    fair_label: str,
    query_text: str,
    image_hint_text: str = "",
    limit: int = EXHIBITION_SEARCH_RESULT_COUNT,
) -> List[dict]:
    fair_slugs = set(resolve_fair_slugs(fair_label))
    query = " ".join([str(query_text or "").strip(), str(image_hint_text or "").strip()]).strip()
    q_tokens = _tokenize_semantic(query)

    scored: List[Tuple[float, dict]] = []
    for row in records:
        if row.get("fair_slug") not in fair_slugs:
            continue
        score = _semantic_score(row, q_tokens, _normalize_key(query))
        if query and score <= 0.0:
            continue
        scored.append((score, row))

    scored.sort(
        key=lambda x: (
            -x[0],
            -(x[1].get("year") or 0),
            str(x[1].get("fair_label") or ""),
            str(x[1].get("gallery_name") or ""),
            str(x[1].get("exhibition_title") or ""),
        )
    )
    return [row for _, row in scored[: max(1, int(limit))]]
